[Authorize] public override async Task<GetResponse> Get(GetRequest request, ServerCallContext context) { if (!TryValidateUser(request.TraderId, context.GetHttpContext().User)) { throw new RpcException( new Status(StatusCode.PermissionDenied, "Denied.")); } var portfolio = await _repository.GetAsync(traderId, request.PortfolioId); return new GetResponse { Portfolio = Portfolio.FromRepositoryModel(portfolio) }; }
