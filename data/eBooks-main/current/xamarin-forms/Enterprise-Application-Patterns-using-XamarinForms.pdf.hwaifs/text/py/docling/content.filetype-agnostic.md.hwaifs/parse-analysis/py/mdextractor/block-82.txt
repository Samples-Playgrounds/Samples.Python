public async Task<TResult> GetAsync<TResult>(string uri, string token = "" ) { HttpClient httpClient = CreateHttpClient(token); HttpResponseMessage response = await httpClient.GetAsync(uri); await HandleResponse(response); string serialized = await response.Content.ReadAsStringAsync(); TResult result = await Task.Run(() => JsonConvert.DeserializeObject<TResult>(serialized, _serializerSettings)); return result; }
