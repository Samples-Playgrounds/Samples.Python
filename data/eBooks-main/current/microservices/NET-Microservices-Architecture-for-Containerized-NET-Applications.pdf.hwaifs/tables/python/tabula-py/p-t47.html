0 /// <summary>
1 /// Base class for implementing a long running <see cref="IHostedService"/>.
2 /// </summary>
3 public abstract class BackgroundService : IHostedService, IDisposable
4 {
5 private Task _executingTask;
6 private readonly CancellationTokenSource _stoppingCts =
7 new CancellationTokenSource();
8
9 protected abstract Task ExecuteAsync(CancellationToken stoppingToken);
10
11 public virtual Task StartAsync(CancellationToken cancellationToken)
12 {
13 // Store the task we're executing
14 _executingTask = ExecuteAsync(_stoppingCts.Token);
15
16 // If the task is completed then return it,
17 // this will bubble cancellation and failure to the caller
18 if (_executingTask.IsCompleted)
19 {
20 return _executingTask;
21 }
22
23 // Otherwise it's running
24 return Task.CompletedTask;
25 }
26
27 public virtual async Task StopAsync(CancellationToken cancellationToken)
28 {
29 // Stop called without start
30 if (_executingTask == null)
31 {
32 return;
33 }
